基本的使用


 
import request from '@/utils/request'

export default {

 
  checkExsis(data) {
    return request({
      url: '/login/checkExsis',
      method: 'get',
      params: { iphone: data }
    })
  },
 
 
  getInfo(token) {
    return request({
      url: '/login/getInfo',
      method: 'get',
      params: { token }
    })
  }
  login(username, password) {
    return request({
      url: '/user/login',
      method: 'post',
      data: {
        username,
        password
      }
    })
  },

}
